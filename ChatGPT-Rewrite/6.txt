# `src/core/reducer.rs`

```rust
use std::sync::Arc;

use crate::core::events::{
    AppEvent,
    MarketEvent,
    OrderEvent,
    TradeEvent,
};

use crate::core::shared_state::SharedState;

pub fn reduce(
    shared: &SharedState,
    event: AppEvent,
) {
    let current = shared.load();

    let mut next = (*current).clone();

    match event {
        AppEvent::Market(event) => {
            apply_market(
                &mut next,
                event,
            );
        }

        AppEvent::Order(event) => {
            apply_order(
                &mut next,
                event,
            );
        }

        AppEvent::Trade(event) => {
            apply_trade(
                &mut next,
                event,
            );
        }

        AppEvent::Ui(_) => {}

        AppEvent::System(_) => {}
    }

    next.version += 1;

    shared.store(
        Arc::new(next)
    );
}

fn apply_market(
    state: &mut AppState,
    event: MarketEvent,
) {
    match event {
        MarketEvent::BidUpdated {
            market_id,
            bid,
        } => {
            if let Some(market) =
                state.markets.get_mut(&market_id)
            {
                market.bid = bid;
            }
        }

        MarketEvent::AskUpdated {
            market_id,
            ask,
        } => {
            if let Some(market) =
                state.markets.get_mut(&market_id)
            {
                market.ask = ask;
            }
        }

        MarketEvent::LastTradeUpdated {
            market_id,
            price,
        } => {
            if let Some(market) =
                state.markets.get_mut(&market_id)
            {
                market.last_price = price;
            }
        }

        _ => {}
    }
}

fn apply_order(
    state: &mut AppState,
    event: OrderEvent,
) {
    match event {
        OrderEvent::OrderUpdated {
            order,
        } => {
            state.orders.insert(
                order.order_id.clone(),
                order,
            );
        }

        OrderEvent::OrderRemoved {
            order_id,
        } => {
            state.orders.remove(
                &order_id
            );
        }

        _ => {}
    }
}

fn apply_trade(
    state: &mut AppState,
    event: TradeEvent,
) {
    match event {
        TradeEvent::TradeUpdated {
            trade,
        } => {
            state.trades.insert(
                trade.trade_id.clone(),
                trade,
            );
        }
    }
}
```

# `src/worker/reducer_task.rs`

```rust
use tokio::sync::mpsc::Receiver;

use crate::core::events::AppEvent;

use crate::core::reducer::reduce;

use crate::core::shared_state::SharedState;

const MAX_BATCH: usize = 256;

pub async fn start_reducer_task(
    shared: SharedState,

    mut rx: Receiver<AppEvent>,
) {
    let mut batch =
        Vec::with_capacity(MAX_BATCH);

    loop {
        batch.clear();

        let first = match rx.recv().await {
            Some(event) => event,

            None => break,
        };

        batch.push(first);

        while let Ok(event) =
            rx.try_recv()
        {
            batch.push(event);

            if batch.len() >= MAX_BATCH {
                break;
            }
        }

        for event in batch.drain(..) {
            reduce(
                &shared,
                event,
            );
        }
    }
}
```

# `src/infrastructure/repaint_scheduler.rs`

```rust
use std::sync::atomic::{
    AtomicBool,
    AtomicU64,
    Ordering,
};

use std::time::{
    Duration,
    SystemTime,
    UNIX_EPOCH,
};

pub struct RepaintScheduler {
    dirty: AtomicBool,

    queued: AtomicBool,

    last_frame_ms: AtomicU64,

    min_frame_interval_ms: u64,
}

impl RepaintScheduler {
    pub fn new(
        min_frame_interval_ms: u64,
    ) -> Self {
        Self {
            dirty: AtomicBool::new(false),

            queued: AtomicBool::new(false),

            last_frame_ms:
                AtomicU64::new(0),

            min_frame_interval_ms,
        }
    }

    pub fn mark_dirty(&self) {
        self.dirty.store(
            true,
            Ordering::Relaxed,
        );
    }

    pub fn should_repaint(
        &self,
    ) -> bool {
        if !self
            .dirty
            .load(Ordering::Relaxed)
        {
            return false;
        }

        let now = current_time_ms();

        let last = self
            .last_frame_ms
            .load(Ordering::Relaxed);

        if now.saturating_sub(last)
            < self.min_frame_interval_ms
        {
            return false;
        }

        self.last_frame_ms.store(
            now,
            Ordering::Relaxed,
        );

        self.dirty.store(
            false,
            Ordering::Relaxed,
        );

        true
    }
}

fn current_time_ms() -> u64 {
    SystemTime::now()
        .duration_since(
            UNIX_EPOCH
        )
        .unwrap_or(Duration::ZERO)
        .as_millis() as u64
}
```

# `src/app/snapshot/snapshot_cache.rs`

```rust
pub struct SnapshotCache {
    last_version: u64,

    snapshot: UiSnapshot,
}

impl SnapshotCache {
    pub fn new() -> Self {
        Self {
            last_version: 0,

            snapshot: UiSnapshot {
                markets: Vec::new(),

                orders: Vec::new(),

                trades: Vec::new(),
            },
        }
    }

    pub fn snapshot(
        &mut self,

        state: &AppState,
    ) -> &UiSnapshot {
        if state.version
            != self.last_version
        {
            self.snapshot =
                UiSnapshot {
                    markets: state
                        .markets
                        .values()
                        .cloned()
                        .collect(),

                    orders: state
                        .orders
                        .values()
                        .cloned()
                        .collect(),

                    trades: state
                        .trades
                        .values()
                        .cloned()
                        .collect(),
                };

            self.last_version =
                state.version;
        }

        &self.snapshot
    }
}
```

```rust
use tokio::sync::mpsc;

use crate::core::events::AppEvent;

pub struct PriorityQueues {
    pub high_priority_tx:
        mpsc::Sender<AppEvent>,

    pub high_priority_rx:
        mpsc::Receiver<AppEvent>,

    pub low_priority_tx:
        mpsc::Sender<AppEvent>,

    pub low_priority_rx:
        mpsc::Receiver<AppEvent>,
}

impl PriorityQueues {
    pub fn new() -> Self {
        let (
            high_priority_tx,
            high_priority_rx,
        ) = mpsc::channel(4096);

        let (
            low_priority_tx,
            low_priority_rx,
        ) = mpsc::channel(16384);

        Self {
            high_priority_tx,

            high_priority_rx,

            low_priority_tx,

            low_priority_rx,
        }
    }
}
```
